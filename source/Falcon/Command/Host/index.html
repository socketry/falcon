Falcon::Command::Host

Implements the falcon host command. Designed for deployment.

Manages a class Falcon::Controller::Host instance which is responsible for running applications in a production environment.

def configuration

Generate a configuration based on the specified Falcon::Command::Host#paths.

Implementation

 
def configuration
	configuration = Configuration.new
	
	@paths.each do |path|
		path = File.expand_path(path)
		configuration.load_file(path)
	end
	
	return configuration
end

def call
	Console.logger.info(self) do |buffer|
		buffer.puts "Falcon Host v#{VERSION} taking flight!"
		buffer.puts "- Configuration: #{@paths.join(', ')}"
		buffer.puts "- To terminate: Ctrl-C or kill #{Process.pid}"
		buffer.puts "- To reload: kill -HUP #{Process.pid}"
	end
	
	begin
		Bundler.require(:preload)
	rescue Bundler::GemfileNotFound
		# Ignore.
	end
	
	if GC.respond_to?(:compact)
		GC.compact
	end
	
	self.controller.run
end
